struct ListNode* mergeTwoLists(struct ListNode* l1, struct ListNode* l2){
    struct ListNode head;
    struct ListNode * pre = &head;

    while(l1&&l2)
    {
        if(l1->val<=l2->val) 
        {
            pre->next=l1;
            l1=l1->next;
        }
        else{
             pre->next=l2;
             l2=l2->next;
        }

        pre=pre->next;
    }
    pre->next = (NULL == l1 ? l2 : l1); //如果l1为空，说明l1已经遍历完。
    return head.next;

}
